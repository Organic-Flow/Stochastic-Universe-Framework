"""
Benchmark 12: Zero-Shot Cross-Modal Resonance
=====================================================================
Proves Cross-Modal Associative Recall without Projection Layer.

Scenario: Pavlov's Dog
  Modality A (8D)  — "Bell Sound"  → s_0..s_7
  Modality B (16D) — "Food Image"  → v_0..v_15

Phases:
  1. Symbiosis Training: intra-modal clustering (FeaturePull) +
     cross-modal Hebbian bridges (same concept → strengthen)
  2. Freeze: trait_strengths frozen — only activation diffuses
  3. Zero-Shot Retrieval: soft-trigger of new (unseen) A query →
     energy wave → which B concept lights up?

Why it is revolutionary:
  Classical: needs 8×16=128 learnable params (Linear Layer)
  OrganicFlow: topological bridges O(N·k), zero matrix multiplications

Metrics:
  top1_accuracy   — correct B concept at the top (target > 0.80)
  energy_ratio    — correct/mean_wrong energy ratio (target > 5x)
  cross_density   — % bridges with strength > 0.3
"""
import sys
import os
sys.path.insert(0, os.path.abspath(os.path.join(os.path.dirname(__file__), "..")))

import json
import random
import time
import tracemalloc
import numpy as np

from OrganicFlow.base.core_v7 import MorphoCell, MorphoField


# ─── Config ───────────────────────────────────────────────────────────────────

SEED          = 42
N_CONCEPTS    = 20      # distinct concepts
CELLS_TRAIN   = 4       # training cells per concept per modality
CELLS_TEST    = 1       # held-out query cells per concept (zero-shot test)
DIM_A         = 8       # Modality A — "sound"
DIM_B         = 16      # Modality B — "vision"
NOISE_PROTO   = 0.12    # within-concept spread
INTRA_EPOCHS  = 15      # Hebbian bridge epochs (no FeaturePull drift)
BRIDGE_LR     = 0.08
WAVE_STEPS    = 3
NEIGHBORS_K   = 5
SOFT_SIGMA    = 0.20     # Gaussian kernel bandwidth for soft-trigger

TRAITS_A = [f"s_{i}" for i in range(DIM_A)]
TRAITS_B = [f"v_{i}" for i in range(DIM_B)]


# ─── Data generation ──────────────────────────────────────────────────────────

def generate_concepts(rng):
    """N_CONCEPTS prototype pairs (8D, 16D) + noisy training/test instances."""
    pa = rng.randn(N_CONCEPTS, DIM_A)
    pb = rng.randn(N_CONCEPTS, DIM_B)
    pa /= np.linalg.norm(pa, axis=1, keepdims=True) + 1e-8
    pb /= np.linalg.norm(pb, axis=1, keepdims=True) + 1e-8

    def _make(protos, dim, n):
        X, y = [], []
        for cid, p in enumerate(protos):
            for _ in range(n):
                v = np.clip((p + rng.randn(dim) * NOISE_PROTO + 1.0) / 2.0, 0.0, 1.0)
                X.append(v)
                y.append(cid)
        return np.array(X, dtype=np.float32), np.array(y, dtype=np.int64)

    Xa_tr, ya_tr = _make(pa, DIM_A, CELLS_TRAIN)
    Xa_te, ya_te = _make(pa, DIM_A, CELLS_TEST)
    Xb_tr, yb_tr = _make(pb, DIM_B, CELLS_TRAIN)
    return Xa_tr, ya_tr, Xa_te, ya_te, Xb_tr, yb_tr


# ─── Field construction ────────────────────────────────────────────────────────

def build_field(X, y, traits, dim):
    field = MorphoField(
        mode="stochastic_wave", neighbors_k=NEIGHBORS_K,
        noise_scale=0.02, wave_steps=6, wave_threshold=0.05, tunneling_prob=0.15,
    )
    labels = {}
    by_concept = {cid: [] for cid in range(N_CONCEPTS)}
    for vec, cid in zip(X, y):
        cell = MorphoCell(state_dim=dim, traits=traits, noise_scale=0.001)
        for i, v in enumerate(vec):
            cell.state.trait_strengths[traits[i]] = float(v)
        field.cells_by_dim[dim].append(cell)
        labels[cell.id] = int(cid)
        by_concept[int(cid)].append(cell)
    return field, labels, by_concept


# ─── Cross-modal bridge management ────────────────────────────────────────────

def init_bridges(by_a, by_b, init_s=0.02):
    """same-concept bridges only: a2b[ca_id][cb_id] = [strength] (mutable)."""
    a2b = {}
    for cid in range(N_CONCEPTS):
        for ca in by_a[cid]:
            a2b.setdefault(ca.id, {})
            for cb in by_b[cid]:
                a2b[ca.id][cb.id] = [init_s]
    return a2b


def hebbian_step(a2b, by_a, by_b, lr):
    for cid in range(N_CONCEPTS):
        for ca in by_a[cid]:
            ea = float(np.mean([ca.state.trait_strengths.get(t, 0.5) for t in TRAITS_A]))
            for cb in by_b[cid]:
                eb = float(np.mean([cb.state.trait_strengths.get(t, 0.5) for t in TRAITS_B]))
                entry = a2b.get(ca.id, {}).get(cb.id)
                if entry is not None:
                    entry[0] = min(1.0, entry[0] + lr * ea * eb)


def bridge_density(a2b, threshold=0.3):
    total = above = 0
    for cb_dict in a2b.values():
        for (s,) in cb_dict.values():
            total += 1
            if s > threshold:
                above += 1
    return round(above / max(total, 1), 4)


# ─── Retrieval ─────────────────────────────────────────────────────────────────

def soft_trigger_wave(qvec, all_a, all_b, a2b, steps, sigma=SOFT_SIGMA):
    """
    Soft-trigger A cells via Gaussian kernel on L2 distance to query,
    then propagate energy through frozen bridges to B cells.
    Gaussian is far more discriminating than cosine when all values ≈ 0.5.
    Trait strengths are NEVER modified — pure activation diffusion.
    """
    act_a = {}
    for ca in all_a:
        v = np.array([ca.state.trait_strengths.get(t, 0.0) for t in TRAITS_A])
        dist_sq = float(np.sum((qvec - v) ** 2))
        act_a[ca.id] = float(np.exp(-dist_sq / (2 * sigma ** 2)))

    # Normalize so max = 1.0 (preserve relative magnitudes)
    max_act = max(act_a.values()) + 1e-8
    act_a = {k: v / max_act for k, v in act_a.items()}

    act_b = {cb.id: 0.0 for cb in all_b}

    for _ in range(steps):
        new_b = dict(act_b)
        for ca in all_a:
            ea = act_a[ca.id]
            if ea < 1e-4:
                continue
            for cb_id, (strength,) in a2b.get(ca.id, {}).items():
                if cb_id in new_b:
                    new_b[cb_id] += ea * strength
        act_a = {k: v * 0.5 for k, v in act_a.items()}
        act_b = new_b

    return act_b


def evaluate_retrieval(by_a, by_b, a2b, all_a, all_b, Xa_te, ya_te):
    b_id2cid = {c.id: cid for cid, cells in by_b.items() for c in cells}
    top1 = top3 = 0
    ratios = []

    for qvec, true_cid in zip(Xa_te, ya_te):
        act_b = soft_trigger_wave(qvec, all_a, all_b, a2b, WAVE_STEPS)

        ce = {}
        for cb_id, act in act_b.items():
            cid = b_id2cid[cb_id]
            ce[cid] = ce.get(cid, 0.0) + act

        ranked = sorted(ce.items(), key=lambda x: x[1], reverse=True)
        if ranked and ranked[0][0] == int(true_cid):
            top1 += 1
        if int(true_cid) in [c for c, _ in ranked[:3]]:
            top3 += 1

        correct = ce.get(int(true_cid), 0.0)
        wrong = [e for cid, e in ce.items() if cid != int(true_cid)]
        mean_wrong = float(np.mean(wrong)) if wrong else 1e-8
        ratios.append(correct / max(mean_wrong, 1e-8))

    n = len(ya_te)
    return {
        "top1_accuracy":     round(top1 / n, 4),
        "top3_accuracy":     round(top3 / n, 4),
        "mean_energy_ratio": round(float(np.mean(ratios)), 4),
        "min_energy_ratio":  round(float(np.min(ratios)), 4),
        "n_queries":         n,
    }


# ─── Baselines ─────────────────────────────────────────────────────────────────

def run_classical(Xa_tr, ya_tr, Xb_tr, yb_tr, Xa_te, ya_te):
    """Ridge Regression A→B projection + nearest-centroid in B space."""
    from sklearn.linear_model import Ridge

    b_by_cid = {}
    for vec, cid in zip(Xb_tr, yb_tr):
        b_by_cid.setdefault(int(cid), []).append(vec)
    centroids = np.array([np.mean(b_by_cid[cid], axis=0) for cid in range(N_CONCEPTS)])

    targets = np.array([centroids[int(cid)] for cid in ya_tr])
    reg = Ridge(alpha=0.1)
    reg.fit(Xa_tr, targets)

    proj = reg.predict(Xa_te)
    hits = sum(
        np.argmin(np.linalg.norm(centroids - p, axis=1)) == int(tc)
        for p, tc in zip(proj, ya_te)
    )
    return {
        "top1_accuracy": round(hits / len(ya_te), 4),
        "note": "Ridge Regression: 8×16 projection matrix → nearest centroid",
    }


def run_random(ya_te):
    rng2 = np.random.RandomState(SEED + 99)
    preds = rng2.randint(0, N_CONCEPTS, size=len(ya_te))
    return {
        "top1_accuracy": round(float(np.mean(preds == ya_te)), 4),
        "note": f"random guess, expected ≈{1/N_CONCEPTS:.3f}",
    }

    print(f"  A: train={Xa_tr.shape}  test={Xa_te.shape}  |  B: train={Xb_tr.shape}")

    tracemalloc.start()
    t0 = time.perf_counter()

    print("\n[2/4] Building modality fields...")
    field_a, labels_a, by_a = build_field(Xa_tr, ya_tr, TRAITS_A, DIM_A)
    field_b, labels_b, by_b = build_field(Xb_tr, yb_tr, TRAITS_B, DIM_B)
    all_a = [c for cells in by_a.values() for c in cells]
    all_b = [c for cells in by_b.values() for c in cells]
    a2b = init_bridges(by_a, by_b)
    n_bridges = sum(len(v) for v in a2b.values())
    print(f"  A: {len(all_a)} cells | B: {len(all_b)} cells | bridges: {n_bridges}")

    print(f"\n[3/4] Symbiosis training ({INTRA_EPOCHS} Hebbian epochs)...")
    for ep in range(INTRA_EPOCHS):
        hebbian_step(a2b, by_a, by_b, BRIDGE_LR)
        if (ep + 1) % 5 == 0 or ep == INTRA_EPOCHS - 1:
            ss = [s[0] for cb_d in a2b.values() for s in cb_d.values()]
            print(f"  epoch {ep+1}/{INTRA_EPOCHS}  "
                  f"bridge_mean={np.mean(ss):.4f}  bridge_max={np.max(ss):.4f}")

    train_ms = round((time.perf_counter() - t0) * 1000, 1)
    _, peak = tracemalloc.get_traced_memory()
    tracemalloc.stop()

    density = bridge_density(a2b)

    print(f"\n[4/4] Zero-shot retrieval (frozen weights, {WAVE_STEPS} wave steps)...")
    print(f"  Cross-modal density (strength>0.3): {density}")

    m_organic   = evaluate_retrieval(by_a, by_b, a2b, all_a, all_b, Xa_te, ya_te)
    m_classical = run_classical(Xa_tr, ya_tr, Xb_tr, yb_tr, Xa_te, ya_te)

          f"top3={m_organic['top3_accuracy']:.4f}  "
          f"energy_ratio={m_organic['mean_energy_ratio']:.2f}x")
    print(f"  Classical top1={m_classical['top1_accuracy']:.4f}  (Ridge Regression 8×16)")
