OrganicFlow Benchmark 01 PASS

Spontaneous Self-Organisation
From Chaos to Structure

OrganicFlow transforms pure noise into organised data structure — with no loss function and no backpropagation. Order emerges exclusively through topological interaction.

The Claim

Given 200 cells initialised with 100% random noise in an 8-dimensional space, OrganicFlow's stochastic_wave engine will spontaneously produce coherent cluster structure — without any supervision, labels, or cost function.

The Problem with Classical Approaches

In traditional Machine Learning, organising data requires either supervised learning (labels, ground truth) or algorithms like K-Means — which demand prior knowledge of the number of clusters k, operate top-down by computing global centroids, and cannot process continuous or infinite data streams.

Neither approach can self-organise. Both require an external architect to define the structure before learning begins.

Experiment Setup
01

Initialisation

200 MorphoCells placed in an 8-dimensional trait space. All initial values are pure random noise — no prior structure, no seeding, no warm start.

02

The Rule

Neutral Interaction only. Cells exchange information with their nearest neighbours via the stochastic_wave propagation engine. No objective. No right or wrong.

03

The Baseline

A standard Random Walk with no network interaction. Cells move independently, with no topological influence on each other.

Performance Visualisation

Note: The aggressive topological mode achieves near-zero zero distance (mode collapse), but at ~65x the latency of our balanced stochastic_wave engine.

Technical Due Diligence
Q

Does the network simply collapse into a single point — Mode Collapse?

A

Under strict topological conditions (k-NN only), yes — this risk exists. In our tests, topological mode reduced mean distance to 0.001501, a near-total collapse. This is precisely why our production architecture is stochastic_wave. It introduces controlled stochastic tunnelling that preserves healthy information diversity (distance 0.935121), preventing collapse while still achieving organisation.

Q

How is this different from K-Means or DBSCAN?

A

K-Means requires static data and global computation — it cannot process new data without a full rerun. OrganicFlow operates strictly locally via message passing. This means it can accept a continuous, infinite stream of new data in real time and reorganise dynamically — something structurally impossible for static clustering algorithms.

Raw Data & Source Code
"""
Benchmark 1: Spontaneous Self-Organization
==========================================
Proves that MorphoField can transform chaos into structure
without PCA/backpropagation — through topological interaction alone.

Runs with 3 modes for comparison:
  - topological      (strict k-NN, collapse risk)
  - stochastic_wave  (OrganicFlow production mode)
  - baseline         (no-interaction random walk)
"""
import sys
import os
sys.path.insert(0, os.path.abspath(os.path.join(os.path.dirname(__file__), '..')))

import time
import tracemalloc
import numpy as np
import random
import json
from OrganicFlow.base.core_v7 import MorphoField, NeutralInteractionRule

SEED      = 42
NUM_CELLS = 200
STATE_DIM = 8
EPOCHS    = 15
RULE      = NeutralInteractionRule(strength=0.15)


def get_trait_vectors(cells):
    return np.array([
        [cell.state.trait_strengths[t] for t in cell.state.traits]
        for cell in cells
    ])


def get_cluster_signature(cells):
    vectors   = get_trait_vectors(cells)
    centroid  = vectors.mean(axis=0)
    distances = np.linalg.norm(vectors - centroid, axis=1)
    threshold = np.median(distances)
    return int(np.sum(distances <= threshold)), int(np.sum(distances > threshold))


def mean_pairwise_distance(cells):
    vectors = get_trait_vectors(cells)
    n = len(vectors)
    if n < 2:
        return 0.0
    diff  = vectors[:, np.newaxis, :] - vectors[np.newaxis, :, :]
    dists = np.linalg.norm(diff, axis=2)
    idx   = np.triu_indices(n, k=1)
    return float(dists[idx].mean())


def _init_field_cells(field):
    """Initialise cells with random traits (chaos) — fixed seed for reproducibility."""
    random.seed(SEED)
    np.random.seed(SEED)
    field.add_cells(NUM_CELLS, state_dim=STATE_DIM, noise_scale=0.5)
    for cell in field.get_all_cells():
        for trait in cell.state.traits:
            cell.state.trait_strengths[trait] = random.random()


def run_organic(mode):
    """Run benchmark with a specific OrganicFlow mode."""
    field = MorphoField(mode=mode, neighbors_k=5)
    _init_field_cells(field)

    cells = field.get_all_cells()
    c0, _ = get_cluster_signature(cells)
    d0    = mean_pairwise_distance(cells)

    tracemalloc.start()
    t0 = time.perf_counter()
    for _ in range(EPOCHS):
        field.propagate_dynamics(RULE)
    elapsed_ms = round((time.perf_counter() - t0) * 1000, 1)
    _, peak_ram = tracemalloc.get_traced_memory()
    tracemalloc.stop()
    ram_mb = round(peak_ram / 1024 / 1024, 3)

    cells_f        = field.get_all_cells()
    c_final, _     = get_cluster_signature(cells_f)
    d_final        = mean_pairwise_distance(cells_f)
    dist_reduction = d0 - d_final
    passed         = bool(dist_reduction > 0)

    return {
        "mode": mode,
        "passed": passed,
        "initial_mean_dist": round(float(d0), 6),
        "final_mean_dist":   round(float(d_final), 6),
        "dist_reduction":    round(float(dist_reduction), 6),
        "initial_close":     c0,
        "final_close":       c_final,
        "elapsed_ms":        elapsed_ms,
        "ram_mb":            ram_mb,
    }


def run_baseline():
    """Classical baseline: independent random walk, no inter-cell interaction."""
    random.seed(SEED)
    np.random.seed(SEED)
    rng    = np.random.RandomState(SEED)
    states = rng.rand(NUM_CELLS, STATE_DIM)

    n    = NUM_CELLS
    diff = states[:, np.newaxis, :] - states[np.newaxis, :, :]
    idx  = np.triu_indices(n, k=1)
    d0   = float(np.linalg.norm(diff, axis=2)[idx].mean())

    tracemalloc.start()
    t0 = time.perf_counter()
    for _ in range(EPOCHS):
        states += rng.randn(NUM_CELLS, STATE_DIM) * 0.05
        states  = np.clip(states, 0.0, 1.0)
    elapsed_ms = round((time.perf_counter() - t0) * 1000, 1)
    _, peak_ram = tracemalloc.get_traced_memory()
    tracemalloc.stop()
    ram_mb = round(peak_ram / 1024 / 1024, 3)

    diff_f        = states[:, np.newaxis, :] - states[np.newaxis, :, :]
    d_final       = float(np.linalg.norm(diff_f, axis=2)[idx].mean())
    dist_reduction = d0 - d_final

    return {
        "mode": "baseline_random_walk",
        "passed": bool(dist_reduction > 0),
        "initial_mean_dist": round(d0, 6),
        "final_mean_dist":   round(d_final, 6),
        "dist_reduction":    round(dist_reduction, 6),
        "elapsed_ms":        elapsed_ms,
        "ram_mb":            ram_mb,
    }


def run_benchmark():
    print("=" * 60)
    print("BENCHMARK 1: Spontaneous Self-Organization")
    print("=" * 60)

    results = {}
    for mode in ("topological", "stochastic_wave"):
        print(f"\n[{mode}]")
        r = run_organic(mode)
        results[mode] = r
        print(f"  dist_reduction={r['dist_reduction']:+.6f}  passed={r['passed']}"
              f"  time={r['elapsed_ms']}ms  ram={r['ram_mb']}MB")

    print("\n[baseline_random_walk]")
    bl = run_baseline()
    results["baseline"] = bl
    print(f"  dist_reduction={bl['dist_reduction']:+.6f}  passed={bl['passed']}"
          f"  time={bl['elapsed_ms']}ms  ram={bl['ram_mb']}MB")

    print("\n-- Summary --")
    print(f"  {'Mode':<20} {'dist_reduction':>15} {'passed':>7} {'ms':>8} {'RAM MB':>8}")
    for key, r in results.items():
        print(f"  {key:<20} {r['dist_reduction']:>+15.6f} {str(r['passed']):>7}"
              f" {r['elapsed_ms']:>8} {r['ram_mb']:>8}")

    return {
        "benchmark": 1,
        "name": "Spontaneous Self-Organization",
        "topological":     results["topological"],
        "stochastic_wave": results["stochastic_wave"],
        "baseline":        results["baseline"],
    }


if __name__ == "__main__":
    result = run_benchmark()
    os.makedirs(os.path.join(os.path.dirname(__file__), "results"), exist_ok=True)
    out_path = os.path.join(os.path.dirname(__file__), "results", "bench1.json")
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(result, f, indent=2, ensure_ascii=False)
    print(f"\nResults saved to: {out_path}")
