"""
Benchmark 5: Mathematical Proof — HOII Dashboard
=================================================
Runs with 3 modes: topological | stochastic_wave | baseline random-walk
"""
import sys, os, time, tracemalloc, random, json
sys.path.insert(0, os.path.abspath(os.path.join(os.path.dirname(__file__), "..")))
import numpy as np
from OrganicFlow.base.core_v7 import MorphoField, NeutralInteractionRule, StabilityOptimizer
from OrganicFlow.base.helping.metrics import coherence, integration, resilience, entropy_energy

SEED=42; NUM_CELLS=60; STATE_DIM=8; EPOCHS=12
RULE=NeutralInteractionRule(strength=0.1)
OPTIMIZER=StabilityOptimizer(lambda_weight=0.4, learning_rate=0.015, steps=5)

def extract_emb(field):
    return np.array([c.state.to_vector_v2() for c in field.get_all_cells()],dtype=np.float32)

def collect_interactions(field):
    return [tuple(round(r.strength,2) for r in c.state.relations[:4]) for c in field.get_all_cells()]

def identity_fn(x): return x

def compute_rm(field):
    emb=extract_emb(field); pm=emb.mean(axis=0); intrs=collect_interactions(field)
    C=float(np.mean(coherence(emb,pm))); I=float(integration(intrs))
    R=float(resilience(emb,identity_fn,noise_std=0.02,trials=3))
    xp=np.clip(emb.mean(axis=1),1e-6,1.0); E=float(np.mean(entropy_energy(xp,complexity_cost=0.1)))
    return (C*I*R)/(E+1e-6)

def run_organic(mode):
    random.seed(SEED); np.random.seed(SEED)
    field=MorphoField(mode=mode,neighbors_k=4)
    field.add_cells(NUM_CELLS,state_dim=STATE_DIM,noise_scale=0.1)
    tracemalloc.start(); t0=time.perf_counter()
    rm_scores=[]
    for _ in range(EPOCHS):
        field.propagate_dynamics(RULE); OPTIMIZER.optimize(field)
        rm_scores.append(round(compute_rm(field),6))
    ms=round((time.perf_counter()-t0)*1000,1); _,pr=tracemalloc.get_traced_memory(); tracemalloc.stop()
    max_rm=max(rm_scores); med_rm=float(np.median(rm_scores)); peak_above=bool(max_rm>med_rm)
    peak_rel=(max_rm-med_rm)/(med_rm+1e-9)
    return {"mode":mode,"passed":bool(peak_above and max_rm>0.55 and peak_rel>0.10),
            "max_rm_score":max_rm,"median_rm_score":med_rm,"final_rm_score":rm_scores[-1],
            "peak_above_median":peak_above,"elapsed_ms":ms,"ram_mb":round(pr/1024/1024,3)}

def run_baseline():
    random.seed(SEED); np.random.seed(SEED)
    rng=np.random.RandomState(SEED); states=rng.rand(NUM_CELLS,STATE_DIM)
    tracemalloc.start(); t0=time.perf_counter()
    rm_scores=[]
    for _ in range(EPOCHS):
        states+=rng.randn(NUM_CELLS,STATE_DIM)*0.1; states=np.clip(states,0.0,1.0)
        emb=states.astype(np.float32); pm=emb.mean(axis=0)
        intrs=[tuple(np.round(states[i,:4],2)) for i in range(NUM_CELLS)]
        C=float(np.mean(coherence(emb,pm))); I=float(integration(intrs))
        R=float(resilience(emb,lambda x:x,noise_std=0.02,trials=3))
        xp=np.clip(emb.mean(axis=1),1e-6,1.0); E=float(np.mean(entropy_energy(xp,complexity_cost=0.1)))
        rm_scores.append(round((C*I*R)/(E+1e-6),6))
    ms=round((time.perf_counter()-t0)*1000,1); _,pr=tracemalloc.get_traced_memory(); tracemalloc.stop()
    max_rm=max(rm_scores); med_rm=float(np.median(rm_scores))
    peak_rel=(max_rm-med_rm)/(med_rm+1e-9)
    return {"mode":"baseline_random_walk","passed":bool(max_rm>0.9 and peak_rel>0.10),
            "max_rm_score":max_rm,"median_rm_score":med_rm,"final_rm_score":rm_scores[-1],
            "peak_above_median":bool(max_rm>med_rm),"elapsed_ms":ms,"ram_mb":round(pr/1024/1024,3)}

def run_benchmark():
    print("="*60); print("BENCHMARK 5: HOII Dashboard"); print("="*60)
    results={}
    for mode in ("topological","stochastic_wave"):
        print(f"\n[{mode}]"); r=run_organic(mode); results[mode]=r
        print(f"  max_rm={r['max_rm_score']:.4f}  passed={r['passed']}  time={r['elapsed_ms']}ms  ram={r['ram_mb']}MB")
    print("[baseline_random_walk]"); bl=run_baseline(); results["baseline"]=bl
    print(f"  max_rm={bl['max_rm_score']:.4f}  passed={bl['passed']}  time={bl['elapsed_ms']}ms")
    return {"benchmark":5,"name":"HOII Dashboard",
            "topological":results["topological"],"stochastic_wave":results["stochastic_wave"],"baseline":results["baseline"]}

if __name__ == "__main__":
    result=run_benchmark()
    os.makedirs(os.path.join(os.path.dirname(__file__),"results"),exist_ok=True)
    out=os.path.join(os.path.dirname(__file__),"results","bench5.json")
    with open(out,"w",encoding="utf-8") as f: json.dump(result,f,indent=2,ensure_ascii=False)
    print(f"\nResults: {out}")
