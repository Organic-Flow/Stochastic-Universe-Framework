The Claim

Traditional Neural Networks fundamentally suffer from Catastrophic Forgetting. Training on a new Task B inherently overwrites and erases the weight matrices learned from Task A. OrganicFlow logically sidesteps this foundational flaw. By inherently distributing knowledge across a decentralized, spatial topology, the network securely learns massive new visual tasks without blindly overwriting its previously established neural engrams.

The Problem with Classical Approaches

The core problem of Catastrophic Forgetting is the primary reason we cannot simply "update" a trained production model (like ChatGPT or Vision transformers) live with new, incoming streaming data.

The mathematical gradient descent required to learn the new "Task B" mechanically overwrites the rigid, shared parameters (the weight matrix W) already holding the historical knowledge of "Task A". The current industry patch is immensely expensive: conducting a massive, full physical Re-training from scratch, compiling both the old and new datasets simultaneously (Data Rehearsal).

"""
Benchmark 11: Split-MNIST — Continual Learning in the Real World
=====================================================================
Proves that OrganicFlow learns Task B (digits 5-9) without
forgetting Task A (digits 0-4) — without backpropagation, without rehearsal.

Split-MNIST Protocol (standard benchmark in Continual Learning):
  Task A: MNIST digits 0-4 (100 cells/class × 5 classes = 500 cells)
  Task B: MNIST digits 5-9 (100 cells/class × 5 classes = 500 cells)
  Evaluation: test set 0-9 (200/class × 10 = 2000 images)

Philosophy (from user feedback):
  • "Brute-force Flatten" from Demo 2 — 784D pixel vectors, pure NumPy.
  • "Readout Classifier" from Demo 1 — frozen organic field → sklearn readout.
  • stochastic_wave mode — O(N·k) message passing, not O(N²).

Modes compared:
  1. OrganicFlow stochastic_wave — the crown jewel (O(N·k))
  2. Classical sklearn MLP       — catastrophic forgetting baseline
  (topological is already validated in bench 1-7; here we focus
   on stochastic_wave vs classical head-to-head)

Key Numbers:
  • retention_A  — accuracy on Test A after Phase B (target organic > 0.70)
  • learning_B   — accuracy on Test B (target > 0.60)
  • forgetting_gap — how much less the organic model forgets vs classical
"""
import sys
import os
sys.path.insert(0, os.path.abspath(os.path.join(os.path.dirname(__file__), "..")))

import gzip
import json
import random
import struct
import time
import tracemalloc
import numpy as np

from OrganicFlow.base.core_v7 import MorphoCell, MorphoField, FeaturePullInteractionRule
from OrganicFlow.base.helping.consolidation import MorphoBlacklistManager

from sklearn.linear_model import LogisticRegression
from sklearn.neural_network import MLPClassifier
from sklearn.metrics import accuracy_score


# ─── Configuration ────────────────────────────────────────────────────────────

SEED            = 42
STATE_DIM       = 784             # 28x28 flattened
CELLS_PER_CLASS = 100             # 100 × 5 = 500 cells per task
TEST_PER_CLASS  = 200
EPOCHS_A        = 10
EPOCHS_B        = 10
NEIGHBORS_K     = 5
LEARNING_RATE   = 0.05
COMPRESSION     = 1.0             # default — stochastic_wave fires sparsely to be cheap
COMPRESSION_TOP = 0.35            # topological is dense O(N²) — keep per-interaction cost light

TASK_A_DIGITS = (0, 1, 2, 3, 4)
TASK_B_DIGITS = (5, 6, 7, 8, 9)

MNIST_DIR = os.path.abspath(os.path.join(os.path.dirname(__file__), "..", ".data", "MNIST", "raw"))


# ─── MNIST Loader (pure-NumPy, από τα τοπικά ubyte αρχεία) ────────────────────

def _load_idx(path):
    """Loads IDX format (MNIST ubyte or ubyte.gz) into numpy array."""
    opener = gzip.open if path.endswith(".gz") else open
    with opener(path, "rb") as f:
        magic, = struct.unpack(">I", f.read(4))
        if magic == 2051:  # images
            n, rows, cols = struct.unpack(">III", f.read(12))
            data = np.frombuffer(f.read(n * rows * cols), dtype=np.uint8)
            return data.reshape(n, rows * cols)
        elif magic == 2049:  # labels
            n, = struct.unpack(">I", f.read(4))
            return np.frombuffer(f.read(n), dtype=np.uint8)
        else:
            raise ValueError(f"Unknown IDX magic: {magic}")


def load_mnist():
    """Returns (X_train, y_train, X_test, y_test) normalized to [0,1]."""
    def _pick(base):
        for cand in (os.path.join(MNIST_DIR, base),
                     os.path.join(MNIST_DIR, base + ".gz")):
            if os.path.exists(cand):
                return cand
        raise FileNotFoundError(f"Not found: {base}")

    X_tr = _load_idx(_pick("train-images-idx3-ubyte")).astype(np.float32) / 255.0
    y_tr = _load_idx(_pick("train-labels-idx1-ubyte")).astype(np.int64)
    X_te = _load_idx(_pick("t10k-images-idx3-ubyte")).astype(np.float32) / 255.0
    y_te = _load_idx(_pick("t10k-labels-idx1-ubyte")).astype(np.int64)
    return X_tr, y_tr, X_te, y_te


def sample_task(X, y, digits, per_class, rng):
    """Selects `per_class` samples per digit, in `digits`."""
    idxs = []
    for d in digits:
        cand = np.where(y == d)[0]
        chosen = rng.choice(cand, size=min(per_class, len(cand)), replace=False)
        idxs.extend(chosen.tolist())
    idxs = np.array(idxs)
    rng.shuffle(idxs)
    return X[idxs], y[idxs]


# ─── Organic Field Helpers ────────────────────────────────────────────────────

TRAITS = [f"pixel_{i}" for i in range(STATE_DIM)]


def inject_cells_from_images(field, X, y, cell_labels):
    """
    Creates MorphoCells from flattened MNIST images (784D vectors).
    Each pixel becomes a trait_strength. Labels are recorded in the dict.
    Returns a list of the newly created cells.
    """
    new_cells = []
    for img, label in zip(X, y):
        cell = MorphoCell(state_dim=STATE_DIM, traits=TRAITS, noise_scale=0.01)
        for i, v in enumerate(img):
            cell.state.trait_strengths[TRAITS[i]] = float(v)
        field.cells_by_dim[STATE_DIM].append(cell)
        cell_labels[cell.id] = int(label)
        new_cells.append(cell)
    return new_cells


def extract_vectors(cells):
    """The trait_strengths (784D) of each cell as an array."""
    return np.array(
        [[c.state.trait_strengths[t] for t in TRAITS] for c in cells],
        dtype=np.float32,
    )


def run_organic_phase(field, cells, cell_labels, epochs, rule, label=""):
    """Runs `epochs` epochs in the field; returns time in ms."""
    t0 = time.perf_counter()
    for ep in range(epochs):
        field.propagate_dynamics(rule)
        if (ep + 1) % 2 == 0 or ep == epochs - 1:
            vecs = extract_vectors(cells)
            std = float(vecs.std(axis=0).mean())
            print(f"    [{label}] epoch {ep+1}/{epochs}  trait_std={std:.4f}")
    return round((time.perf_counter() - t0) * 1000, 1)


def train_organic(mode, X_a, y_a, X_b, y_b, X_test, y_test):
    """
    Full organic pipeline:
      1. Phase A: injects Task A cells, runs epochs_a with feature_pull.
      2. Blacklist A cells (for transparent documentation).
      3. Phase B: adds Task B cells to the SAME field, runs epochs_b.
      4. Freeze → readout → accuracy on Test A / Test B / Overall.
    """
    random.seed(SEED)
    np.random.seed(SEED)

    # stochastic_wave: increase wave_steps & k_seeds for denser coverage of the 500 cells
    field_kwargs = dict(mode=mode, neighbors_k=NEIGHBORS_K, noise_scale=0.02)
    if mode == "stochastic_wave":
        field_kwargs.update(wave_steps=6, wave_threshold=0.05, tunneling_prob=0.15)
    field = MorphoField(**field_kwargs)
    cell_labels: dict = {}

    tracemalloc.start()
    t_all = time.perf_counter()

    # ── Phase A ───────────────────────────────────────────────────────────
    print(f"  Phase A: injecting {len(X_a)} cells (digits 0-4)...")
    cells_a = inject_cells_from_images(field, X_a, y_a, cell_labels)
    compression = COMPRESSION_TOP if mode == "topological" else COMPRESSION
    rule = FeaturePullInteractionRule(
        cell_labels=cell_labels,
        learning_rate=LEARNING_RATE,
        compression_rate=compression,
    )
    ms_a = run_organic_phase(field, cells_a, cell_labels, EPOCHS_A, rule, "A")

    # Accuracy after only-A (pre-B readout) — to see how much it learned A
    acc_a_pre, _ = readout_accuracy(cells_a, cell_labels, X_test, y_test,
                                    keep_classes=TASK_A_DIGITS, tag="A-pre")

    # Blacklist all A (documentation · protection is inherent to the rule)
    blacklist = MorphoBlacklistManager()
    for c in cells_a:
        blacklist.blacklist(c)

    # ── Phase B ───────────────────────────────────────────────────────────
    print(f"  Phase B: adding {len(X_b)} cells (digits 5-9) to the SAME field...")
    cells_b = inject_cells_from_images(field, X_b, y_b, cell_labels)
    # The rule sees the updated cell_labels (same reference maintained).
    ms_b = run_organic_phase(field, cells_a + cells_b, cell_labels, EPOCHS_B, rule, "B")

    # ── Readout (LogReg primary, MLP secondary) ───────────────────────────
    all_cells = cells_a + cells_b
    acc_A_post, acc_B_post, acc_overall, secondary = full_readout(
        all_cells, cell_labels, X_test, y_test
    )

    elapsed_ms = round((time.perf_counter() - t_all) * 1000, 1)
    _, peak = tracemalloc.get_traced_memory()
    tracemalloc.stop()

    return {
        "mode": mode,
        "acc_A_after_taskA":  round(acc_a_pre, 4),
        "acc_A_after_taskB":  round(acc_A_post, 4),
        "acc_B_after_taskB":  round(acc_B_post, 4),
        "acc_overall":        round(acc_overall, 4),
        "retention_A":        round(acc_A_post / max(acc_a_pre, 1e-8), 4),
        "forgetting_A":       round(max(0.0, acc_a_pre - acc_A_post), 4),
        "mlp_secondary":      secondary,
        "phase_A_ms":         ms_a,
        "phase_B_ms":         ms_b,
        "total_ms":           elapsed_ms,
        "peak_ram_mb":        round(peak / 1024 / 1024, 2),
        "n_cells":            len(all_cells),
        "failsafe_triggers":  field.failsafe_triggers,
    }


# ─── Readout (Frozen organic field → sklearn classifier) ──────────────────────

def readout_accuracy(cells, cell_labels, X_test, y_test, keep_classes, tag=""):
    """
    Trains LogisticRegression on the trait_strengths of the frozen cells
    and returns accuracy on the test set (filtered on `keep_classes`).
    """
    X_train = extract_vectors(cells)
    y_train = np.array([cell_labels[c.id] for c in cells], dtype=np.int64)

    # filter the test on kept classes
    mask = np.isin(y_test, np.array(keep_classes))
    X_te, y_te = X_test[mask], y_test[mask]

    clf = LogisticRegression(max_iter=500, solver="lbfgs")
    clf.fit(X_train, y_train)
    acc = accuracy_score(y_te, clf.predict(X_te))
    print(f"    [readout:{tag}] LogReg acc={acc:.4f}  (test_n={len(y_te)})")
    return float(acc), clf


def full_readout(cells, cell_labels, X_test, y_test):
    """Runs LogReg (primary) + MLP (secondary) and splits into A/B/Overall."""
    X_train = extract_vectors(cells)
    y_train = np.array([cell_labels[c.id] for c in cells], dtype=np.int64)

    clf = LogisticRegression(max_iter=500, solver="lbfgs")
    clf.fit(X_train, y_train)
    preds = clf.predict(X_test)

    mask_A = np.isin(y_test, np.array(TASK_A_DIGITS))
    mask_B = np.isin(y_test, np.array(TASK_B_DIGITS))
    acc_A = accuracy_score(y_test[mask_A], preds[mask_A])
    acc_B = accuracy_score(y_test[mask_B], preds[mask_B])
    acc_all = accuracy_score(y_test, preds)
    print(f"    [readout:FULL] LogReg | A={acc_A:.4f} | B={acc_B:.4f} | all={acc_all:.4f}")

    # Secondary: MLP — more expressive, for robustness check
    mlp = MLPClassifier(
        hidden_layer_sizes=(128,), max_iter=120, random_state=SEED,
        early_stopping=True, validation_fraction=0.1,
    )
    mlp.fit(X_train, y_train)
    mlp_preds = mlp.predict(X_test)
    mlp_acc_A = accuracy_score(y_test[mask_A], mlp_preds[mask_A])
    mlp_acc_B = accuracy_score(y_test[mask_B], mlp_preds[mask_B])
    mlp_acc_all = accuracy_score(y_test, mlp_preds)
    print(f"    [readout:FULL] MLP    | A={mlp_acc_A:.4f} | B={mlp_acc_B:.4f} | all={mlp_acc_all:.4f}")

    secondary = {
        "mlp_acc_A": round(float(mlp_acc_A), 4),
        "mlp_acc_B": round(float(mlp_acc_B), 4),
        "mlp_acc_overall": round(float(mlp_acc_all), 4),
    }
    return float(acc_A), float(acc_B), float(acc_all), secondary


# ─── Classical Baseline (sklearn MLP, sequential training) ────────────────────

def run_classical_baseline(X_a, y_a, X_b, y_b, X_test, y_test):
    """
    Classical Catastrophic Forgetting scenario:
      1. Train MLP on Task A (warm_start=True, using partial_fit for continual).
      2. Continue training ONLY on Task B (no rehearsal/replay of A).
      3. Measure how much the accuracy drops on Test A.
    """
    random.seed(SEED)
    np.random.seed(SEED)

    t0 = time.perf_counter()
    mask_A = np.isin(y_test, np.array(TASK_A_DIGITS))
    mask_B = np.isin(y_test, np.array(TASK_B_DIGITS))

    # An MLP that will see A first, then B — this is the textbook catastrophic
    # forgetting setup. We use partial_fit to explicitly control the
    # phases (no rehearsal).
    all_classes = np.array(sorted(set(TASK_A_DIGITS) | set(TASK_B_DIGITS)))
    mlp = MLPClassifier(
        hidden_layer_sizes=(128,), learning_rate_init=0.01,
        max_iter=1, random_state=SEED, solver="sgd",
    )

    # Phase A — 15 passes (so it converges before the drift)
    for _ in range(15):
        mlp.partial_fit(X_a, y_a, classes=all_classes)
    acc_A_pre = accuracy_score(y_test[mask_A], mlp.predict(X_test[mask_A]))

    # Phase B — continual training ONLY on B
    for _ in range(15):
        mlp.partial_fit(X_b, y_b, classes=all_classes)

    preds = mlp.predict(X_test)
    acc_A = accuracy_score(y_test[mask_A], preds[mask_A])
    acc_B = accuracy_score(y_test[mask_B], preds[mask_B])
    acc_all = accuracy_score(y_test, preds)
    elapsed_ms = round((time.perf_counter() - t0) * 1000, 1)

    print(f"    [classical] A_pre={acc_A_pre:.4f} → A_post={acc_A:.4f}  B={acc_B:.4f}  all={acc_all:.4f}")

    return {
        "mode": "classical_mlp_sequential",
        "acc_A_after_taskA":  round(float(acc_A_pre), 4),
        "acc_A_after_taskB":  round(float(acc_A), 4),
        "acc_B_after_taskB":  round(float(acc_B), 4),
        "acc_overall":        round(float(acc_all), 4),
        "retention_A":        round(float(acc_A / max(acc_A_pre, 1e-8)), 4),
        "forgetting_A":       round(float(max(0.0, acc_A_pre - acc_A)), 4),
        "total_ms":           elapsed_ms,
        "note":               "sklearn MLP partial_fit, Phase A 15 passes → Phase B 15 passes (no rehearsal).",
    }


# ─── Main ─────────────────────────────────────────────────────────────────────

def run_benchmark():
    print("=" * 64)
    print("BENCHMARK 11: Split-MNIST — Continual Learning (Organic vs Classical)")
    print("=" * 64)
    print(f"  Task A: digits {TASK_A_DIGITS}  ({CELLS_PER_CLASS}/class)")
    print(f"  Task B: digits {TASK_B_DIGITS}  ({CELLS_PER_CLASS}/class)")
    print(f"  Test:   {TEST_PER_CLASS}/class × 10")
    print(f"  Field:  STATE_DIM={STATE_DIM}, k={NEIGHBORS_K}, epochs_A={EPOCHS_A}, epochs_B={EPOCHS_B}")

    # ── Dataset prep ─────────────────────────────────────────────────────
    print("\n[1/4] Loading MNIST from local ubyte files...")
    X_tr, y_tr, X_te, y_te = load_mnist()
    print(f"  train: {X_tr.shape}, test: {X_te.shape}")

    rng = np.random.RandomState(SEED)
    X_a, y_a = sample_task(X_tr, y_tr, TASK_A_DIGITS, CELLS_PER_CLASS, rng)
    X_b, y_b = sample_task(X_tr, y_tr, TASK_B_DIGITS, CELLS_PER_CLASS, rng)
    X_test, y_test = sample_task(X_te, y_te, TASK_A_DIGITS + TASK_B_DIGITS,
                                 TEST_PER_CLASS, rng)
    print(f"  sampled: A={X_a.shape}, B={X_b.shape}, test={X_test.shape}")

    # ── Organic: stochastic_wave ─────────────────────────────────────────
    print("\n[2/3] Organic mode: stochastic_wave")
    res_sw = train_organic("stochastic_wave", X_a, y_a, X_b, y_b, X_test, y_test)

    # ── Classical baseline ───────────────────────────────────────────────
    print("\n[3/3] Classical MLP sequential baseline")
    res_cls = run_classical_baseline(X_a, y_a, X_b, y_b, X_test, y_test)

    # ── Summary table ────────────────────────────────────────────────────
    print("\n" + "=" * 64)
    print("SUMMARY (Retention_A = acc_A_post / acc_A_pre — closer to 1.0 means less forgetting)")
    print("=" * 64)
    print(f"  {'mode':<28} {'A_pre':>6} {'A_post':>7} {'B':>6} {'all':>6} {'ret_A':>7} {'forget':>7}")
    for r in (res_sw, res_cls):
        print(f"  {r['mode']:<28}"
              f" {r['acc_A_after_taskA']:>6.3f}"
              f" {r['acc_A_after_taskB']:>7.3f}"
              f" {r['acc_B_after_taskB']:>6.3f}"
              f" {r['acc_overall']:>6.3f}"
              f" {r['retention_A']:>7.3f}"
              f" {r['forgetting_A']:>7.3f}")

    # passed criteria
    organic_best = res_sw["retention_A"]
    classical_ret = res_cls["retention_A"]
    # Target: organic forgets significantly less than classical
    passed = bool(
        organic_best > classical_ret + 0.15
        and res_sw["acc_B_after_taskB"] > 0.30
    )
    print(f"\n  Organic best retention_A : {organic_best:.4f}")
    print(f"  Classical retention_A    : {classical_ret:.4f}")
    print(f"  PASS (organic > classical+0.15 & B>0.30): {passed}")

    return {
        "benchmark": 11,
        "name": "Split-MNIST — Continual Learning (Organic vs Classical)",
        "passed": passed,
        "config": {
            "seed": SEED,
            "state_dim": STATE_DIM,
            "cells_per_class": CELLS_PER_CLASS,
            "test_per_class": TEST_PER_CLASS,
            "epochs_a": EPOCHS_A,
            "epochs_b": EPOCHS_B,
            "neighbors_k": NEIGHBORS_K,
            "learning_rate": LEARNING_RATE,
            "compression_rate": COMPRESSION,
            "task_a": list(TASK_A_DIGITS),
            "task_b": list(TASK_B_DIGITS),
        },
        "stochastic_wave": res_sw,
        "classical":       res_cls,
        "summary": {
            "organic_best_retention_A": round(organic_best, 4),
            "classical_retention_A":    round(classical_ret, 4),
            "retention_advantage":      round(organic_best - classical_ret, 4),
        },
    }
