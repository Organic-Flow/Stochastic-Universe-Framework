The Problem with the Status Quo

Connecting fundamentally disparate data modalities (like Text, Image, and Audio) is the Holy Grail of modern AI. Standard models like CLIP or DALL-E achieve this by training from scratch on massive, paired datasets using expensive techniques like Contrastive Learning (InfoNCE). This costs millions of dollars in compute time.

The core question is: what if you already have two excellent, independently trained models? How do you marry them without retraining from scratch? The standard approach of using a Projection Layer usually fails to capture the complexity of the spaces.

# ─── Load caches ──────────────────────────────────────────────────────────────

def load_caches():
    paths = {
        "img_emb":    os.path.join(CACHE_DIR, "img_emb.npy"),
        "img_labels": os.path.join(CACHE_DIR, "img_labels.npy"),
        "text_emb":   os.path.join(CACHE_DIR, "text_emb.npy"),
        "text_labels":os.path.join(CACHE_DIR, "text_labels.npy"),
        "class_names":os.path.join(CACHE_DIR, "class_names.json"),
    }
    for k, p in paths.items():
        if not os.path.exists(p):
            raise FileNotFoundError(
                f"Cache '{p}' not found.\n"
                "Run first:  python flow_testing/embedders/extract_embeddings.py"
            )
    img_emb    = np.load(paths["img_emb"]).astype(np.float32)
    img_labels = np.load(paths["img_labels"]).astype(np.int64)
    text_emb   = np.load(paths["text_emb"]).astype(np.float32)
    text_labels= np.load(paths["text_labels"]).astype(np.int64)
    with open(paths["class_names"]) as f:
        class_names = json.load(f)
    return img_emb, img_labels, text_emb, text_labels, class_names


# ─── Split images into train / test ──────────────────────────────────────────

def split_images(img_emb, img_labels, imgs_train, imgs_test, rng):
    train_X, train_y, test_X, test_y = [], [], [], []
    for cid in range(N_CLASSES):
        idx = np.where(img_labels == cid)[0]
        rng.shuffle(idx)
        n_tr = min(imgs_train, len(idx) - imgs_test)
        n_te = min(imgs_test,  len(idx) - n_tr)
        for i in idx[:n_tr]:
            train_X.append(img_emb[i]); train_y.append(cid)
        for i in idx[n_tr:n_tr + n_te]:
            test_X.append(img_emb[i]);  test_y.append(cid)
    return (np.array(train_X, dtype=np.float32), np.array(train_y, dtype=np.int64),
            np.array(test_X,  dtype=np.float32), np.array(test_y,  dtype=np.int64))


# ─── Adaptive sigma (scales with √D) ─────────────────────────────────────────

def adaptive_sigma(vecs, labels, scale=0.25, rng=None):
    """
    Compute Gaussian bandwidth from mean within-class L2 distance x scale.
    Using overall median gives sigma ~ cross-class dist -> no discrimination.
    Within-class mean x 0.25 gives tight, discriminative kernel.
    """
    if rng is None:
        rng = np.random.RandomState(SEED)
    same_dists = []
    classes = np.unique(labels)
    for cid in classes:
        idx = np.where(labels == cid)[0]
        if len(idx) < 2:
            continue
        chosen = rng.choice(idx, size=min(8, len(idx)), replace=False)
        sub = vecs[chosen]
        for i in range(len(sub)):
            for j in range(i + 1, len(sub)):
                same_dists.append(float(np.linalg.norm(sub[i] - sub[j])))
    return float(np.mean(same_dists)) * scale


# ─── Field construction ────────────────────────────────────────────────────────

def norm01(X):
    """Per-dimension min-max normalization to [0.05, 0.95].
    L2-normalized embeddings have mean~0 — without this, trait_strengths~0
    and Hebbian lr*ea*eb ~ 0 forever."""
    lo = X.min(axis=0); hi = X.max(axis=0)
    return ((X - lo) / (hi - lo + 1e-8) * 0.90 + 0.05).astype(np.float32)


def build_field_from_matrix(X, y, traits, dim):
    """Create MorphoCells for each embedding vector, returns (field, by_class).
    Normalizes embeddings to [0.05,0.95] so Hebbian activations are non-zero."""
    X = norm01(X)
    field = MorphoField(
        mode="stochastic_wave", neighbors_k=NEIGHBORS_K,
        noise_scale=0.01, wave_steps=6, wave_threshold=0.05, tunneling_prob=0.10,
    )
    by_class = {cid: [] for cid in range(N_CLASSES)}
    for vec, cid in zip(X, y):
        cell = MorphoCell(state_dim=dim, traits=traits, noise_scale=0.001)
        for i, v in enumerate(vec):
            cell.state.trait_strengths[traits[i]] = float(v)
        field.cells_by_dim[dim].append(cell)
        by_class[int(cid)].append(cell)
    return field, by_class, X  # return normalized X for retrieval use


# ─── Cross-modal bridges ──────────────────────────────────────────────────────

def init_bridges(by_a, by_b, init_s=0.02):
    """same-class A<->B bridges; a2b[a_id][b_id] = [strength]."""
    a2b = {}
    for cid in range(N_CLASSES):
        for ca in by_a[cid]:
            a2b.setdefault(ca.id, {})
            for cb in by_b[cid]:
                a2b[ca.id][cb.id] = [init_s]
    return a2b


def hebbian_step(a2b, by_a, by_b, lr, traits_a, traits_b):
    """Hebbian update: δs prop mean_a x mean_b.
    Works correctly now that embeddings are normalized to [0.05, 0.95]."""
    for cid in range(N_CLASSES):
        a_acts = np.array([
            np.mean([ca.state.trait_strengths[t] for t in traits_a])
            for ca in by_a[cid]
        ])
        b_acts = np.array([
            np.mean([cb.state.trait_strengths[t] for t in traits_b])
            for cb in by_b[cid]
        ])
        for ai, ca in enumerate(by_a[cid]):
            for bi, cb in enumerate(by_b[cid]):
                entry = a2b.get(ca.id, {}).get(cb.id)
                if entry is not None:
                    entry[0] = min(1.0, entry[0] + lr * float(a_acts[ai]) * float(b_acts[bi]))


def bridge_density(a2b, threshold=0.3):
    total = above = 0
    for cb_dict in a2b.values():
        for (s,) in cb_dict.values():
            total += 1
            if s > threshold:
                above += 1
    return round(above / max(total, 1), 4)


# ─── Retrieval ─────────────────────────────────────────────────────────────────

def soft_trigger_wave(qvec, all_b, a2b_flat, steps, sigma):
    """
    qvec: 512D image query (in B space).
    Soft-trigger B cells via Gaussian, propagate through bridges to A.

    We trigger B->A (reverse direction):
      - Gaussian activation on B cells from image query
      - energy flows B->A through transposed bridges

    Returns dict {a_cell_id: activation}
    """
    # B activation via Gaussian kernel
    act_b = {}
    for cb in all_b:
        v = np.array([cb.state.trait_strengths.get(t, 0.0) for t in TRAITS_B],
                     dtype=np.float32)
        dist_sq = float(np.sum((qvec - v) ** 2))
        act_b[cb.id] = float(np.exp(-dist_sq / (2 * sigma ** 2)))

    max_act = max(act_b.values()) + 1e-8
    act_b = {k: v / max_act for k, v in act_b.items()}

    # Propagate B->A through transposed bridges
    act_a = {}
    for _ in range(steps):
        new_a = dict(act_a)
        for ca_id, cb_dict in a2b_flat.items():
            for cb_id, strength in cb_dict.items():
                eb = act_b.get(cb_id, 0.0)
                if eb < 1e-5:
                    continue
                new_a[ca_id] = new_a.get(ca_id, 0.0) + eb * strength
        act_b = {k: v * 0.5 for k, v in act_b.items()}
        act_a = new_a

    return act_a


def evaluate_retrieval(by_a, all_b, a2b, test_X, test_y, sigma):
    """
    For each test image (512D), soft-trigger B cells, propagate to A,
    aggregate by class, return top1/top5 accuracy and energy ratio.
    """
    # Build flat bridge lookup: {a_cell_id: {b_cell_id: strength}}
    a2b_flat = {
        ca_id: {cb_id: s[0] for cb_id, s in cb_dict.items()}
        for ca_id, cb_dict in a2b.items()
    }

    # Map A cell ids -> class
    a_id2cid = {c.id: cid for cid, cells in by_a.items() for c in cells}
    all_a_ids = list(a_id2cid.keys())

    top1 = top5 = 0
    ratios = []

    for qvec, true_cid in zip(test_X, test_y):
        act_a = soft_trigger_wave(qvec, all_b, a2b_flat, WAVE_STEPS, sigma)

        # aggregate by class
        ce = {}
        for a_id, act in act_a.items():
            cid = a_id2cid.get(a_id)
            if cid is not None:
                ce[cid] = ce.get(cid, 0.0) + act

        # class energy for classes with no A-activation defaults to 0
        ranked = sorted(ce.items(), key=lambda x: x[1], reverse=True)
        top_ids = [c for c, _ in ranked]

        if top_ids and top_ids[0] == int(true_cid):
            top1 += 1
        if int(true_cid) in top_ids[:5]:
            top5 += 1

        correct   = ce.get(int(true_cid), 0.0)
        wrong_vals = [e for cid, e in ce.items() if cid != int(true_cid)]
        mean_wrong = float(np.mean(wrong_vals)) if wrong_vals else 1e-8
        ratios.append(correct / max(mean_wrong, 1e-8))

    n = len(test_y)
    return {
        "top1_accuracy":     round(top1 / n, 4),
        "top5_accuracy":     round(top5 / n, 4),
        "mean_energy_ratio": round(float(np.mean(ratios)),  4),
        "min_energy_ratio":  round(float(np.min(ratios)),   4),
        "n_queries":         n,
    }


# ─── Learned Projection Baseline ──────────────────────────────────────────────

def run_projection_baseline(text_emb, text_labels, train_X, train_y, test_X, test_y):
    """
    Learns a linear W: 384->512 with InfoNCE contrastive loss.
    Paired (text, image) samples from same class are pulled together.
    ~196K learnable parameters.
    """
    try:
        import torch
        import torch.nn as nn
        import torch.optim as optim
    except ImportError:
        return {"error": "torch not installed", "top1_accuracy": None}

    rng2 = np.random.RandomState(SEED + 1)

    # Build paired training set: for each text, pick a random same-class image
    img_by_class = {}
    for vec, cid in zip(train_X, train_y):
        img_by_class.setdefault(int(cid), []).append(vec)

    txt_by_class = {}
    for vec, cid in zip(text_emb, text_labels):
        txt_by_class.setdefault(int(cid), []).append(vec)

    pairs_t, pairs_i = [], []
    for cid in range(N_CLASSES):
        if cid not in img_by_class or cid not in txt_by_class:
            continue
        for tvec in txt_by_class[cid]:
            ivec = img_by_class[cid][rng2.randint(len(img_by_class[cid]))]
            pairs_t.append(tvec)
            pairs_i.append(ivec)

    T = torch.tensor(np.array(pairs_t, dtype=np.float32))
    I = torch.tensor(np.array(pairs_i, dtype=np.float32))

    W = nn.Linear(DIM_A, DIM_B, bias=False)
    nn.init.xavier_uniform_(W.weight)
    optimizer = optim.Adam(W.parameters(), lr=BASELINE_LR)
    n_params = sum(p.numel() for p in W.parameters())

    BATCH = 256
    t0 = time.perf_counter()
    for epoch in range(BASELINE_EPOCHS):
        perm = torch.randperm(len(T))
        total_loss = 0.0
        for b_start in range(0, len(T), BATCH):
            idx = perm[b_start:b_start + BATCH]
            t_batch = T[idx]
            i_batch = I[idx]
            t_proj = W(t_batch)
            t_norm = t_proj / (t_proj.norm(dim=-1, keepdim=True) + 1e-8)
            i_norm = i_batch / (i_batch.norm(dim=-1, keepdim=True) + 1e-8)
            sim = torch.matmul(t_norm, i_norm.T) / BASELINE_TEMP
            labels = torch.arange(len(idx))
            loss = (nn.functional.cross_entropy(sim, labels) +
                    nn.functional.cross_entropy(sim.T, labels)) / 2
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            total_loss += loss.item()
        if (epoch + 1) % 10 == 0:
            print(f"    [projection] epoch {epoch+1}/{BASELINE_EPOCHS}  loss={total_loss:.4f}")

    train_ms = round((time.perf_counter() - t0) * 1000, 1)

    # Evaluation: project test images, retrieve via text class centroids
    with torch.no_grad():
        # Text class centroids in projected space
        text_t = torch.tensor(text_emb)
        text_proj = W(text_t)
        text_proj_np = text_proj.numpy()
        centroids = np.zeros((N_CLASSES, DIM_B), dtype=np.float32)
        counts = np.zeros(N_CLASSES, dtype=np.int64)
        for vec, cid in zip(text_proj_np, text_labels):
            centroids[cid] += vec
            counts[cid] += 1
        for cid in range(N_CLASSES):
            if counts[cid] > 0:
                centroids[cid] /= counts[cid]

        # Nearest centroid retrieval in image space (no projection for images)
        top1 = top5 = 0
        for qvec, true_cid in zip(test_X, test_y):
            dists = np.linalg.norm(centroids - qvec, axis=1)
            ranked = np.argsort(dists)
            if ranked[0] == int(true_cid):
                top1 += 1
            if int(true_cid) in ranked[:5]:
                top5 += 1

    n = len(test_y)
    return {
        "top1_accuracy":  round(top1 / n, 4),
        "top5_accuracy":  round(top5 / n, 4),
        "n_params":       n_params,
        "train_ms":       train_ms,
        "note":           f"Linear W: {DIM_A}x{DIM_B}={n_params} params, InfoNCE contrastive, {BASELINE_EPOCHS} epochs",
    }


def run_random_baseline(test_y):

    preds = rng2.randint(0, N_CLASSES, size=len(test_y))
    return {
        "top1_accuracy": round(float(np.mean(preds == test_y)), 4),

    }

    t0 = time.perf_counter()

    all_a = [c for cells in by_a.values() for c in cells]
    all_b = [c for cells in by_b.values() for c in cells]
    a2b   = init_bridges(by_a, by_b)
    n_bridges = sum(len(v) for v in a2b.values())

    print(f"  A (text):  {len(all_a)} cells ({DIM_A}D)")
    print(f"  B (image): {len(all_b)} cells ({DIM_B}D)")
    print(f"  Bridges:   {n_bridges:,}  (same-class only)")

    # ── [3] Adaptive sigma (within-class, on normalized training images) ──
    sigma_b = adaptive_sigma(img_train_norm, img_train_y, scale=0.25, rng=rng)
    print(f"\n  Adaptive sigma (B, {DIM_B}D, within-class*0.25): {sigma_b:.4f}")

    # ── [4] Hebbian training ──────────────────────────────────────────────
    print(f"\n[3/5] Hebbian Symbiosis Training ({INTRA_EPOCHS} epochs)...")
    for ep in range(INTRA_EPOCHS):
        hebbian_step(a2b, by_a, by_b, BRIDGE_LR, TRAITS_A, TRAITS_B)
        if (ep + 1) % 5 == 0 or ep == INTRA_EPOCHS - 1:
            ss = [s[0] for cb_d in a2b.values() for s in cb_d.values()]
            print(f"  epoch {ep+1}/{INTRA_EPOCHS}  "
                  f"bridge_mean={np.mean(ss):.4f}  bridge_max={np.max(ss):.4f}")

    train_ms = round((time.perf_counter() - t0) * 1000, 1)
    _, peak = tracemalloc.get_traced_memory()
    tracemalloc.stop()

    density = bridge_density(a2b)
    print(f"  Cross-modal density (strength>0.3): {density}")

    # ── [5] Retrieval ─────────────────────────────────────────────────────
    print(f"\n[4/5] Zero-Shot Retrieval ({len(img_test_y)} test images, frozen weights)...")
    m_organic = evaluate_retrieval(by_a, all_b, a2b, img_test_norm, img_test_y, sigma_b)
